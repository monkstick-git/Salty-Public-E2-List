@name FireSpitter
@inputs Rpm
@outputs T1 T2 T3 T4 T5 T6
@persist Time
@trigger 
interval(100)

if(first()|duped()){
 Time = 0.3   
}

if(Rpm >= 2300){
if(random(10) <Time ){T1=1} else{T1 = 0}
if(random(10) <Time ){T2=1} else{T2 = 0}
if(random(10) <Time ){T3=1} else{T3 = 0}
if(random(10) <Time ){T4=1} else{T4 = 0}
if(random(10) <Time ){T5=1} else{T5 = 0}
if(random(10) <Time ){T6=1} else{T6 = 0}
}
else{
T1=0
T2=0
T3=0
T4=0
T5=0
T6=0
}
