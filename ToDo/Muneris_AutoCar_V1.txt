@name Muneris AutoCar V1
@inputs RPM W
@outputs Throttle Gear
@persist Timer ShiftAmount
@trigger 
runOnTick(1)
if(first()|duped()){
Timer = 0
ShiftAmount = 20
Gears = 8
Max = 5000
Min = 3000
}

if(tickClk()){
    if(RPM > Max){
        Timer = Timer + ShiftAmount        
    }    
    
    if(Timer >= 100){
        Gear = Gear + 1
        Timer = 0   
    }
}
