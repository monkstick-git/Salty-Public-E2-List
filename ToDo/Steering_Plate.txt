@name Steering Plate
@inputs A D Chassis:entity
@outputs Angle:angle
@persist Turn P Y R Ang
@model models/sprops/rectangles/size_2/rect_12x12x3.mdl

E=entity()
runOnTick(1)
E:propGravity(0)
Turn=A-D
if(A)
{
 Ang = clamp(Ang + 3,-30,30)   
}
if(D)
{
 Ang = clamp(Ang - 3,-30,30)   
}
Angle=Chassis:angles()
P=Angle:pitch()
Y=Angle:yaw()
R=Angle:roll()
E:setAng(Angle)
E:propFreeze(1)
if(!A & !D){E:setAng(Angle)}
else{
    E:setAng(ang(P,Y+Ang,R))
}
