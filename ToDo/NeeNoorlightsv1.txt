@name NeeNoorlightsv1
@inputs 
@outputs A:vector B:vector C:vector D:vector Sound On
@persist N Sound2 SS
@trigger 
interval(1000)
runOnTick(1)
N=N+10
SS = SS + 1
Light1 = abs(sin(N))*255

A = vec(Light1,0,255-Light1)
B = vec(Light1,0,255-Light1)
C = vec(Light1,0,255-Light1)
D = vec(Light1,0,255-Light1)

if(SS > 33)
{
 SS = 0
Sound2 = !Sound2 
}

Sound = (Sound2 * 1.49) + 1
On = 1
