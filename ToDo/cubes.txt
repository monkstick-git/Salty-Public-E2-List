@name cubes
@inputs 
@outputs 
@persist 
@trigger 
interval(1)
if(first() | duped()){
 holoCreate(1)
holoCreate(2)  

holoScale(1,vec(4,4,4))
holoScale(2,vec(5,5,5))
 
}

holoPos(1,entity():pos() + vec(0,0,150))
holoPos(2,entity():pos() + vec(0,0,150))

holoColor(1,vec(255,0,255))
holoColor(2,vec(0,255,255))
holoAlpha(2,254)

for(I=1,20){
 holoCreate(I)
 holoScale(I,vec(I,I,I))
holoColor(I,vec(I,100,50 + I))
holoAlpha(I,254)   
}
