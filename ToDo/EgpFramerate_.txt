@name EgpFramerate 
@inputs EGP:wirelink Active
@outputs A
@persist A B
@trigger 

interval(100)
if(first()){
 EGP:egpBox(1,vec2(256,256),vec2(512,512)) 
EGP:egpBox(2,vec2(256,256),vec2(256,30))  
A=1
}

if(Active){
A=!A
EGP:egpBox(1,vec2(256,256),vec2(A*512,A*512))
}else{EGP:egpBox(1,vec2(256,256),vec2(A,A))}
B=B+66
EGP:egpAngle(2,B)
