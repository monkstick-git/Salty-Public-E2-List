@name DeltaSpeed
@inputs 
@outputs A
@persist OldSpeed Speedvel CalcSpeed B
@trigger 
interval(250)

Speedvel = entity():vel():length()
CalcSpeed = Speedvel - OldSpeed
entity():setText(floor(CalcSpeed):toString())
OldSpeed = Speedvel

if(CalcSpeed < -300)
{
 B=1   
}

if(B)
{
 A = !A   
 B=B+1
}

if(B >=25){
 B = 0
 A = 0   
}
