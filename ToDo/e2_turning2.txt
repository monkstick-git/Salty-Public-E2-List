@name e2 turning2
@inputs A D
@outputs 
@persist 
@trigger 
@model models/hunter/plates/plate1x1.mdl

runOnTick(1)
if(tickClk()){
TurnAngle = 30

E = entity()
Base = E:isConstrainedTo("axis")
Local = ang(0, TurnAngle*(A-D) - 90, 0) - Base:toLocal(E:angles())        
DesAF = (Local*300 - E:angVel()*45) * shiftL(ang(E:inertia()))        
E:applyAngForce( DesAF * ang(0,1,0))
}
