@name Masswire
@inputs 
@outputs 
@persist
@trigger 

if (first()) {
    runOnChat(1)
}

if (chatClk(owner())) {
    ShouldWire = 0
    Input = ""
    if (lastSaid() == "-active") {
        ShouldWire = 1
        Input = "Active"
    }
    if (lastSaid() == "-throttle") {
        ShouldWire = 1
        Input = "Throttle"
    }
    if (ShouldWire) {
        WireTo = owner():aimEntity()
        if (WireTo:isValid()) {
            findIncludePlayerProps(owner())
            findByClass("acf_engine")
            Boxes = findToArray()
            Count = Boxes:count()
            
            for (K=1,Count) {
                Box = Boxes[K,entity]
                Box:createWire(WireTo,Input,"Throttle")
            }
        }
    }
}
