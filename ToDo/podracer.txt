@name podracer
@inputs W A S D
@outputs L R
@persist 
@trigger 

if(W == 1 & A == 0 & S == 0 & D == 0){
L = 1
R = 1
}
else{
 L = 0
R = 0   
}


if(W == 0 & A == 1 & S == 0 & D == 0){
 L = 0
R = 1   
}
else{
 L = 0
R = 0   
}


if(W == 0 & A == 0 & S == 0 & D == 1){
 L = 1
R = 0  
}
else{
 L = 0
R = 0   
}


if(W == 0 & A == 0 & S == 1 & D == 0){
 L = -1
R = -1
}
else{
 L = 0
R = 0   
}

if(W & A & S & D == 0){
 L = 0
R = 0   
}
