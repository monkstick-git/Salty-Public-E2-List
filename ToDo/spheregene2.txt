@name SphereGenE2
@outputs X Y Z
@persist Theda Phi A
interval(1)
if(first()){
    for(I=1,1){
 holoCreate(I) 
holoEntity(1):setTrails(10000,0,10,"trails/physbeam",vec(255,0,0),255) 
}
}

Theda += 0.07
Phi += 0.01
X = 50*cosr(Theda)*sinr(Phi) * 10
Y = 50*sinr(Theda)*sinr(Phi) * 10
Z = 50*cosr(Phi) * 10
if(Phi > pi()){Phi = 0}
if(Theda > 2*pi()){Theda = 0}

#A=A+1
holoPos(1,entity():pos() + vec(X,Y,Z)+vec(0,0,1000))
