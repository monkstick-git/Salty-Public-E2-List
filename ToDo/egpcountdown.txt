@name egpcountdown
@inputs EGP:wirelink
@outputs 
@persist Time
@trigger 
interval(100)
if(first()|duped()){
 Time = 15  
EGP:egpSize(2,72) 
}
EGP:egpClear()
Time = Time - 1
if(Time <= 0){
 Time = 0   
}
if(Time > 0){
EGP:egpText(2,Time:toString(),vec2(200,200))
}
else{
EGP:egpText(2,"GO!",vec2(200,200))
EGP:egpColor(2,vec(0,255,0))
EGP:egpSize(2,400)
}
EGP:egpText(1,"CountDown",vec2(210,50))

EGP:egpSize(2,150)

for(I=1,Time){
 EGP:egpCircle(10+I,vec2(25,530 -(I * 50)),vec2(20,20))
}
