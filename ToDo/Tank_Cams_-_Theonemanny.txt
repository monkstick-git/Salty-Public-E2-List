@name Tank Cams - Theonemanny
@inputs Weapon:entity Toggle Enabled
@outputs CamVec:vector CamAng:angle Active
@persist Active Should
@trigger 

runOnTick(1)
if(tickClk())
{
   if(first() | duped())
    {
        Should = 0
    } 
    
    
 CamVec = Weapon:toWorld(vec(50,0,20))
CamAng = Weapon:toWorld(ang(0,0,0))    
    
    if(Enabled == 0){
     Active = 0   
    }
    else{
     Active = Should * Enabled   
    }
}

if((Toggle & ~Toggle)) {
    if(Enabled == 1){
 Should = (!Should)
}
else{
 Active = 0   
}
}
