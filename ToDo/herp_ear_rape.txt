@outputs

interval(200)

A =A+1

soundPlay(1,0,"instruments/bass.wav")
}

soundPlay(2,0,"instruments/hi_hat.wav")

}


if(A >=4){
soundPlay(2,0,"instruments/snare.wav")

}

if(A > 4){
 soundPlay(2,0,"instruments/hi_hat.wav")

}

}
