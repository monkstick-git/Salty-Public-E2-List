@name TurningTank
@inputs Left Right Space
@outputs Hyd Hyd2 Weld
@persist 
@trigger 


if(Left==1 & Space == 1){Hyd=1}
if(Right==1 & Space == 1){Hyd=-1}
if(Left==1 & Space == 1){Hyd2=-1}
if(Right==1 & Space == 1){Hyd2=1}
if(Left ==0 & Right==0){Hyd=0}
if(Left ==0 & Right==0){Hyd2=0}
if(Space == 1){Weld = 0}else{Weld = 1}
