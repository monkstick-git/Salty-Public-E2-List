@name Ballistic calculator
@inputs 
@outputs 
@persist Vel Angle Vx Viy Vfy T D Grav
@trigger 
interval(1)
if(first()|duped()){
Vel=33
holoCreate(1)
holoModel(1,"hq_sphere")
holoScale(1,vec(0.2,0.2,0.2))
entity():isWeldedTo():setAng(ang(-30,0,0))
V = 33
Grav = toUnit("m/s",gravity())
}
Angle = -entity():isWeldedTo():angles():pitch()
Vx = 33 * sin(30)
Viy = 33*cos(30)
Vfy = -Viy

T = -(Vfy - Viy)/Grav
D = toUnit("u",Vx*T)

entity():setText(Grav:toString())
holoPos(1,entity():pos() + vec(D,0,0))
