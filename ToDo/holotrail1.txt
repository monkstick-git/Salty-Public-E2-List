@name HoloTrail1
@inputs 
@outputs Screen
@persist A
@trigger 
interval(10)
holoCreate(100)
holoModel(100,"cone")
holoScale(100,vec(5,5,5))
holoPos(100,entity():pos() + vec(0,0,-40))
A = A + 1
holoCreate(A)
holoModel(A,"sphere")
holoScale(A,vec(5,5,5))
holoPos(A,entity():pos() + vec(0,0,sin(A) * 100))
holoColor(A,vec(random(255),random(255),random(255)))

holoDelete(A - 10)  
if(A >= 10){
 A = 0   
}

Screen = holoRemainingSpawns()
